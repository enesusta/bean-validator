    @Min(min = 5)

    @Min(min = 5)
    private short aShort;

    @Min(min = 6)
    private int anInt;

    @Min(min = 5)
    private long aLong;

    @Min(min = 5)

    @Min(min = 5)

    }

    public void setaFloat(float aFloat) {

    }

}
